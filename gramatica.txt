musica: declaracoes execucao
declaracoes: {declaracao}
declaracao: declara_acorde | declara_frase
declara_acorde: 'acorde' IDENT acorde
acorde: '{' {nota} '}'
nota: base acidente oitava
base: 'A' | 'B' | 'C' | 'D' | 'E' | 'F' | 'G'
acidente: '#' | 'b' | <<vazio>>
oitava: NUM_INT
declara_frase: 'frase' IDENT frase
frase: '{' {evento} '}'
evento: evento_tempo | frase_ident
evento_tempo: (nota | acorde_ident | 'pausa') '(' duracao ')'
acorde_ident: acorde | IDENT
frase_ident: frase | IDENT
duracao: NUM_INT
execucao: {evento}
